# Semantic Color Tokens

Семантические цвета — это то, что дизайнеры и разработчики используют в компонентах.
Каждый токен переключается автоматически между Light и Dark темой.

export const SemanticRow = ({ name, cssVar, light, dark }) => (
  <div style={{
    display: 'grid',
    gridTemplateColumns: '1fr 100px 100px',
    alignItems: 'center',
    gap: '12px',
    padding: '8px 0',
    borderBottom: '1px solid #f0f0f0',
  }}>
    <div>
      <div style={{ fontWeight: 600, fontSize: '13px' }}>{name}</div>
      <code style={{ fontSize: '11px', color: '#888' }}>{cssVar}</code>
    </div>
    <div style={{
      width: '48px', height: '48px', borderRadius: '8px',
      backgroundColor: light, border: '1px solid #e0e0e0',
    }} />
    <div style={{
      width: '48px', height: '48px', borderRadius: '8px',
      backgroundColor: dark, border: '1px solid #333',
      background: `${dark}`,
      outline: dark === '#ffffff' || dark === '#ffffff80' || dark === '#ffffff40' ? '1px solid #ccc' : 'none',
    }} />
  </div>
);
